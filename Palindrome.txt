package com.technoeleveate.constructor;

public class Palindrome {

	public static void main(String[] args) {
     int n=15551;
     int temp=n;
     int rem;
     int rev=0;
     while(n!=0) {
    	 rem=n%10;
    	 rev=rev*10+rem;
    	 n=n/10;
    	   
     }
     if(temp==rev) {
    	 System.out.println(temp+" is a Palindrome");
     }
     else {
    	 System.out.println(temp+" is not a Palindrome");
     }
	}
}
